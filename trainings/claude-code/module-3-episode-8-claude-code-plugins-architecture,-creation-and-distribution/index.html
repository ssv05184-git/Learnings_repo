üß≠ Quick Overview

The Big Picture:
Plugins are persistent extensions to Claude Code. Unlike one-off automations, plugins become part of Claude's toolkit permanently. They're discoverable, versioned, reusable across projects, and distributable to teams.

What Problem Does This Solve? - Repeated solutions need to be re-implemented
- Team knowledge isn't captured in reusable form
- Custom automation is project-specific and scattered
- Sharing solutions across teams is manual

Where You'll Use This: - Company-specific coding standards
- Domain expertise automation
- Cross-team best practices
- Organizational scaling

üîë Core Concepts Explained

Concept 1: Plugin Architecture

Definition:
Plugins are modular packages containing commands, skills, MCP servers, and configurations that extend Claude Code.

Why It Matters:
Modularity enables composition. Complex capabilities are built from simple plugins.

When To Use:
Whenever you build repeatable automation worth sharing.

Concept 2: Discovery & Registration

Definition:
Plugins are discoverable through registries. Claude discovers and activates them automatically.

Why It Matters:
Discovery eliminates manual setup. Users don't need to know about plugins; Claude finds what's needed.

When To Use:
Essential for adoption and scale.

Concept 3: Distribution & Versioning

Definition:
Plugins are versioned, distributed through registries, and updated independently.

Why It Matters:
Decoupled versioning enables rapid updates without breaking workflows.

When To Use:
For any plugin used by multiple teams or projects.

Part 3: Plugin Implementation

// src/index.ts

import { Plugin, Context, Command } from '@anthropic/claude-code';

export default class CodeReviewPlugin implements Plugin {
  id = 'org.example.code-review';
  name = 'Code Review Enhancement';
  version = '1.0.0';

  commands: Command[] = [];

  async activate(context: Context) {
    console.log('üîç Code Review Plugin Activated');

    // Register commands
    this.registerCommands(context);

    // Register event listeners
    context.on('file-changed', this.onFileChanged.bind(this));
    context.on('pr-created', this.onPRCreated.bind(this));
  }

  private registerCommands(context: Context) {
    context.registerCommand('review-code', async (args) => {
      const { path } = args;

      return await this.reviewCode(path, context);
    });

    context.registerCommand('security-scan', async (args) => {
      const { path } = args;

      return await this.securityScan(path, context);
    });
  }

  private async reviewCode(path: string, context: Context): Promise<string> {
    console.log(`üìã Reviewing code at ${path}`);

    // Read files
    const files = await context.workspace.readDirectory(path);

    // Conduct review using Claude
    const review = await context.claude.review(files, {
      style: context.config.get('review-style'),
      includeSecurityScan: context.config.get('include-security-scan')
    });

    // Send to Slack if configured
    if (context.config.get('integration.slack-webhook')) {
      await this.notifySlack(review, context);
    }

    return review;
  }

  private async securityScan(path: string, context: Context): Promise<string> {
    console.log(`üîí Security scanning ${path}`);

    const files = await context.workspace.readDirectory(path);

    return await context.claude.securityScan(files, {
      level: 'comprehensive'
    });
  }

  private async onFileChanged(event: Event, context: Context) {
    // Optionally run review on file changes
    if (context.config.get('review-on-change')) {
      await this.reviewCode(event.path, context);
    }
  }

  private async onPRCreated(event: Event, context: Context) {
    console.log('üîç Automatically reviewing PR');

    // Access PR information via GitHub MCP
    const prFiles = await context.github.getPRFiles(event.prNumber);

    const review = await context.claude.review(prFiles);

    // Comment on PR
    await context.github.createComment(event.prNumber, review);
  }

  private async notifySlack(review: string, context: Context) {
    const webhook = context.config.get('integration.slack-webhook');

    await fetch(webhook, {
      method: 'POST',
      body: JSON.stringify({
        text: 'Code Review Complete',
        blocks: [{
          type: 'section',
          text: { type: 'mrkdwn', text: review }
        }]
      })
    });
  }
}

Part 4: Testing

// tests/plugin.test.ts

import { describe, it, expect, beforeEach } from '@jest/globals';
import CodeReviewPlugin from '../src/index';
import { MockContext } from '@anthropic/claude-code/testing';

describe('Code Review Plugin', () => {
  let plugin: CodeReviewPlugin;
  let context: MockContext;

  beforeEach(async () => {
    plugin = new CodeReviewPlugin();
    context = new MockContext();
    await plugin.activate(context);
  });

  it('should register review-code command', () => {
    expect(context.commands.has('review-code')).toBe(true);
  });

  it('should perform code review', async () => {
    const result = await context.commands.run('review-code', {
      path: 'src/'
    });

    expect(result).toContain('Review Summary');
  });

  it('should detect security issues', async () => {
    const maliciousCode = `
      eval(userInput);  // Security issue
      password = "hardcoded";  // Security issue
    `;

    context.workspace.mockFile('test.js', maliciousCode);

    const result = await context.commands.run('security-scan', {
      path: 'test.js'
    });

    expect(result).toContain('eval');
    expect(result).toContain('hardcoded');
  });

  it('should notify Slack on completion', async () => {
    context.config.set('integration.slack-webhook', 'https://hooks.slack.com/...');

    const notified = jest.fn();
    context.http.mockPost('**/hooks.slack.com/**', notified);

    await context.commands.run('review-code', { path: 'src/' });

    expect(notified).toHaveBeenCalled();
  });
});
