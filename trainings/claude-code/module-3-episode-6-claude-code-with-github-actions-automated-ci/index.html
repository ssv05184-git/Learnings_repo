Pattern 2: Self-Healing Workflows

name: Self-Healing CI/CD

on:
  push:
    branches: [main]
  schedule:
    - cron: '0 2 * * *'  # Daily check
  workflow_run:
    workflows: ['CI']
    types: [failed]

jobs:
  check-and-heal:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Run diagnostic
        id: diagnose
        run: |
          # Run tests to find issues
          npm test --json > test-results.json || true

          # Parse failures
          FAILURES=$(jq '.failures' test-results.json)
          echo "failures=$FAILURES" >> $GITHUB_OUTPUT

      - name: Claude fixes issues
        if: steps.diagnose.outputs.failures != '0'
        run: |
          # Use Claude to fix detected issues
          claude "
          Fix the failing tests identified in test-results.json:
          $(cat test-results.json)
          "

      - name: Verify fix
        run: npm test

      - name: Create PR if needed
        if: github.event_name == 'workflow_run'
        run: |
          # If changes were made, create a PR
          if [ -n "$(git status --short)" ]; then
            git config user.name "Claude Code Bot"
            git config user.email "claude@anthropic.com"
            git add -A
            git commit -m "fix: Claude auto-fixed failing tests"
            git push origin HEAD:refs/heads/claude-fixes-$(date +%s)
          fi

Pattern 3: Claude with GitHub API

#!/usr/bin/env python3
# claude_github_automation.py

import os
import json
from anthropic import Anthropic

class ClaudeGitHubAutomation:
    def __init__(self):
        self.client = Anthropic()

    def generate_workflow(self, repo_path: str) -> str:
        """Have Claude generate GitHub Actions workflow."""

        # Read project structure
        structure = self._analyze_project(repo_path)

        # Ask Claude to generate workflow
        response = self.client.messages.create(
            model="claude-opus",
            max_tokens=4000,
            messages=[{
                "role": "user",
                "content": f"""
Analyze this project structure and generate a GitHub Actions workflow:

{structure}

Requirements:
- Include linting, testing, and building
- Add deployment steps
- Use best practices
- Include environment variables and secrets

Return only the YAML workflow content.
"""
            }]
        )

        return response.content[0].text

    def _analyze_project(self, repo_path: str) -> str:
        """Analyze project structure."""
        structure = []

        for root, dirs, files in os.walk(repo_path):
            # Skip hidden directories
            dirs[:] = [d for d in dirs if not d.startswith('.')]

            level = root.replace(repo_path, '').count(os.sep)
            indent = ' ' * 2 * level
            structure.append(f'{indent}{os.path.basename(root)}/')

            subindent = ' ' * 2 * (level + 1)
            for file in files[:5]:  # Limit files
                if not file.startswith('.'):
                    structure.append(f'{subindent}{file}')

        return '\n'.join(structure[:100])  # Limit output

# Usage
if __name__ == "__main__":
    automation = ClaudeGitHubAutomation()
    workflow = automation.generate_workflow(".")

    # Write workflow file
    os.makedirs(".github/workflows", exist_ok=True)
    with open(".github/workflows/claude-generated.yml", "w") as f:
        f.write(workflow)

    print("Workflow generated: .github/workflows/claude-generated.yml")

Pattern 4: Intelligent Workflow Optimization

#!/bin/bash
# workflow_optimizer.sh

# Analyze workflow performance
analyze_workflows() {
  echo "Analyzing workflow performance..."

  # Get workflow runs from GitHub API
  RUNS=$(curl -s \
    -H "Authorization: token $GITHUB_TOKEN" \
    "https://api.github.com/repos/$GITHUB_REPOSITORY/actions/runs?per_page=10" \
    | jq '.')

  # Pass to Claude for analysis
  claude "
  Analyze these GitHub Actions runs and suggest optimizations:

  $RUNS

  Suggest:
  1. Caching opportunities
  2. Parallelization improvements
  3. Timeout issues
  4. Expensive operations
  5. Cost optimizations
  "
}

# Implement suggestions
implement_optimizations() {
  echo "Implementing Claude's suggestions..."

  # Claude modifies workflow files
  claude "
  Update .github/workflows/*.yml to:
  1. Add caching for dependencies
  2. Parallelize test and lint jobs
  3. Use pinned action versions
  4. Add timeouts
  5. Remove redundant steps
  "
}

# Run analysis and implement
analyze_workflows
implement_optimizations
